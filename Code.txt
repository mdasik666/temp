import unicodedata

def symbol_to_word_with_unicodedata(text):
    words = []
    for char in text:
        try:
            # Get the Unicode name of the character
            name = unicodedata.name(char)
            words.append(name.lower().replace(" ", "_"))  # Convert to a friendly format
        except ValueError:
            # If no Unicode name exists (e.g., for regular letters), keep the character as is
            words.append(char)
    return " ".join(words)

# Example usage
input_text = "The area of a circle is πr² and the resistance is 10Ω."
converted_text = symbol_to_word_with_unicodedata(input_text)
print(converted_text)
